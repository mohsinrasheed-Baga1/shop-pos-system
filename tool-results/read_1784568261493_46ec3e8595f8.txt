    81тЖТ    81тЖТ * Resize/compress an image file to a base64 data URL using a canvas.
    82тЖТ    82тЖТ * Used for the shop logo so we don't store huge base64 strings in the DB.
    83тЖТ    83тЖТ */
    84тЖТ    84тЖТfunction resizeImageToDataUrl(file: File, maxSize = 200): Promise<string> {
    85тЖТ    85тЖТ  return new Promise((resolve, reject) => {
    86тЖТ    86тЖТ    const reader = new FileReader();
    87тЖТ    87тЖТ    reader.onload = () => {
    88тЖТ    88тЖТ      const img = new Image();
    89тЖТ    89тЖТ      img.onload = () => {
    90тЖТ    90тЖТ        let { width, height } = img;
    91тЖТ    91тЖТ        if (width > height) {
    92тЖТ    92тЖТ          if (width > maxSize) {
    93тЖТ    93тЖТ            height = Math.round((height * maxSize) / width);
    94тЖТ    94тЖТ            width = maxSize;
    95тЖТ    95тЖТ          }
    96тЖТ    96тЖТ        } else {
    97тЖТ    97тЖТ          if (height > maxSize) {
    98тЖТ    98тЖТ            width = Math.round((width * maxSize) / height);
    99тЖТ    99тЖТ            height = maxSize;
   100тЖТ   100тЖТ          }
   101тЖТ   101тЖТ        }
   102тЖТ   102тЖТ        const canvas = document.createElement("canvas");
   103тЖТ   103тЖТ        canvas.width = width;
   104тЖТ   104тЖТ        canvas.height = height;
   105тЖТ   105тЖТ        const ctx = canvas.getContext("2d");
   106тЖТ   106тЖТ        if (!ctx) return reject(new Error("Canvas not supported"));
   107тЖТ   107тЖТ        ctx.drawImage(img, 0, 0, width, height);
   108тЖТ   108тЖТ        try {
   109тЖТ   109тЖТ          resolve(canvas.toDataURL("image/png"));
   110тЖТ   110тЖТ        } catch {
   111тЖТ   111тЖТ          resolve(canvas.toDataURL("image/jpeg", 0.9));
   112тЖТ   112тЖТ        }
   113тЖТ   113тЖТ      };
   114тЖТ   114тЖТ      img.onerror = () => reject(new Error("Image load failed"));
   115тЖТ   115тЖТ      img.src = reader.result as string;
   116тЖТ   116тЖТ    };
   117тЖТ   117тЖТ    reader.onerror = () => reject(new Error("File read failed"));
   118тЖТ   118тЖТ    reader.readAsDataURL(file);
   119тЖТ   119тЖТ  });
   120тЖТ   120тЖТ}

   122тЖТ   122тЖТfunction formatBytes(bytes: number): string {
   123тЖТ   123тЖТ  if (bytes < 1024) return `${bytes} B`;
   124тЖТ   124тЖТ  if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(1)} KB`;
   125тЖТ   125тЖТ  return `${(bytes / (1024 * 1024)).toFixed(2)} MB`;
   126тЖТ   126тЖТ}
